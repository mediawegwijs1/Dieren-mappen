function card(a){

function normalize(s){ return (s||'').toLowerCase().trim().replace(/\s+/g,' '); }

function validateName(input){
  const folderId = input.dataset.folder;
  const def = FOLDERS.find(f=>f.id===folderId);
  const ok = def.answers.some(a => normalize(a) === normalize(input.value));
  const state = document.getElementById('nameState-'+folderId);
  if(ok){
    input.classList.remove('err'); input.classList.add('ok');
    input.disabled = true;
    state.textContent = '✓ goed!';
    namesOK[folderId] = true;
    const folderBox = document.getElementById('folder-'+folderId);
    folderBox.classList.add('ok');
    setTimeout(()=>folderBox.classList.remove('ok'), 900);
    checkAllNamed();
  }else{
    input.classList.remove('ok'); input.classList.add('err');
    state.textContent = 'Niet juist, probeer opnieuw.';
    namesOK[folderId] = false;
  }
}

function checkAllNamed(){
  allReady = FOLDERS.every(f => namesOK[f.id]);
  if(allReady) enableDragPhase();
}

function enableDragPhase(){
  document.querySelectorAll('.card').forEach(el => {
    el.classList.remove('disabled');
    el.draggable = true;
  });
  checkBtn.disabled = false;
  resultBtn.disabled = false;

  document.querySelectorAll('.drop, #bank').forEach(zone => {
    zone.addEventListener('dragover', e=>{ if(!allReady) return; e.preventDefault(); zone.classList.add('hl'); });
    zone.addEventListener('dragleave', ()=> zone.classList.remove('hl'));
    zone.addEventListener('drop', e=>{
      if(!allReady) return;
      e.preventDefault();
      const id = e.dataTransfer.getData('text/plain');
      const el = document.getElementById('card-'+id);
      zone.classList.remove('hl');
      zone.appendChild(el);
      const zoneFolder = zone.dataset.folder;
      if(zoneFolder){ placed[id] = zoneFolder; } else { delete placed[id]; }
      updateStats(false);
    });
  });
}

function build(){
  bank.innerHTML = '';

  shuffled.forEach(a => bank.appendChild(card(a)));
  placed = {}; updateStats(false);

  foldersWrap.innerHTML = '';
  FOLDERS.forEach(f => foldersWrap.appendChild(folderBox(f)));
  namesOK = {}; allReady = false;
  checkBtn.disabled = true; resultBtn.disabled = true;

  buildKeyboard();
}

function updateStats(showAccuracy){
  const placedCount = Object.keys(placed).length;
  leftEl.textContent = (total - placedCount).toString();
  let correct=0;
  for(const [id, folderId] of Object.entries(placed)){
    const def = ANIMALS.find(a=>a.id===id);
    if(def && def.folder === folderId){ correct++; }
  }
  scoreEl.textContent = `${correct} / ${total}`;
  if(showAccuracy){
    document.querySelectorAll('.folder').forEach(f=>f.classList.remove('ok','err'));
    const byFolder = {};
    for(const [id, folderId] of Object.entries(placed)){
      byFolder[folderId] = byFolder[folderId] || {ok:0, total:0};
      const def = ANIMALS.find(a=>a.id===id);
      byFolder[folderId].total++;
      if(def && def.folder===folderId) byFolder[folderId].ok++;
    }
    for(const f of FOLDERS){
      const box = document.getElementById('folder-'+f.id);
      const stats = byFolder[f.id];
      if(!stats) continue;
      box.classList.add(stats.ok===stats.total ? 'ok' : 'err');
      setTimeout(()=>box.classList.remove('ok','err'), 1200);
    }
  }
}

function keyBtn(label, extraClass, onClick){
  const k=document.createElement('div'); k.className='key '+(extraClass||'');
  k.textContent=label; k.style.flex= extraClass==='space' ? '4' : '1.6';
  if(!extraClass) k.style.flex='1';
  k.style.padding='14px 0'; k.style.textAlign='center'; k.style.borderRadius='12px'; k.style.border='1px solid #e5e7eb'; k.style.background='#f8fafc'; k.style.fontWeight='900'; k.style.cursor='pointer'; k.style.userSelect='none'; k.style.fontSize='18px';
  k.addEventListener('click', onClick); return k;
}
function typeKey(ch){
  if(!activeInput){
    const first = Array.from(document.querySelectorAll('.nameRow input')).find(i=>!i.disabled);
    if(first){ first.focus(); activeInput = first; } else { return; }
  }
  if(ch==='BACKSPACE'){
    activeInput.value = activeInput.value.slice(0,-1);
  }else if(ch==='ENTER'){
    validateName(activeInput);
  }else if(ch==='CLEAR'){
    activeInput.value = '';
  }else{
    activeInput.value += (''+ch).toLowerCase();
  }
  activeInput.focus();
}

function showResult(){

    if(userFolder === a.folder){ correct++; }
    else { wrong.push({animal:a, userFolder}); }

  resScore.textContent = `${correct} / ${total}`;
  resPlaced.textContent = placedCount.toString();

  const pct = Math.round((correct/total)*100);
  let msg = '';
  if(placedCount===0){
    msg = 'Je hebt nog geen dieren geplaatst. Probeer eerst te slepen!';
  }else if(pct===100){
    msg = 'Perfect! Alle dieren staan in het juiste leefgebied. 🏆';
  }else if(pct>=70){
    msg = 'Knap! De meeste dieren staan goed. Kijk nog even naar de laatste dieren.';
  }else{
    msg = 'Goed bezig! Je kunt je score nog verbeteren door opnieuw te kijken naar de gebieden.';
  }
  resultSummary.textContent = msg;

  // Build wrong list
  function folderName(id){ return FOLDERS.find(f=>f.id===id)?.name || id; }

    wrongList.innerHTML = '<div class="wrong-item">Geen fouten – alles is geplaatst en correct!</div>';
  }else{
    wrong.forEach(({animal, userFolder}) => {
      const row = document.createElement('div');
      row.className = 'wrong-item';
      row.innerHTML = `<span style="font-size:20px">${animal.emoji}</span>
        <strong>${animal.name}</strong>
        <span style="color:#334155">staat in</span> <span class="badge">${folderName(userFolder)}</span>
        <span style="color:#334155">→ moet naar</span> <span class="badge">${folderName(animal.folder)}</span>`;
      wrongList.appendChild(row);
    });
    notPlaced.forEach(animal => {
      const row = document.createElement('div');
      row.className = 'wrong-item';
      row.innerHTML = `<span style="font-size:20px">${animal.emoji}</span>
        <strong>${animal.name}</strong>
        <span style="color:#334155">is</span> <span class="badge">niet geplaatst</span>
        <span style="color:#334155">→ moet naar</span> <span class="badge">${folderName(animal.folder)}</span>`;
      wrongList.appendChild(row);
    });
