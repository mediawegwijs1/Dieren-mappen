<script>
const FOLDERS = [
  { id:'oceaan', name:'Oceaan', icon:'🌊', answers:['oceaan'] },
  { id:'jungle', name:'Jungle', icon:'🌴', answers:['jungle'] }
];

const ANIMALS = [
  {id:'walvis', name:'Walvis', emoji:'🐋', folder:'oceaan'},
  {id:'dolfijn', name:'Dolfijn', emoji:'🐬', folder:'oceaan'},
  {id:'tijger', name:'Tijger', emoji:'🐅', folder:'jungle'},
  {id:'gorilla', name:'Gorilla', emoji:'🦍', folder:'jungle'}
];

let placed = {};
let activeInput = null;
let namesOK = {};
let allReady = false;

const bank = document.getElementById('bank');
const foldersWrap = document.getElementById('folders');

function card(a){
  const el = document.createElement('div');
  el.className = 'card disabled';
  el.draggable = false;
  el.id = 'card-'+a.id;
  el.textContent = a.emoji + ' ' + a.name;

  el.addEventListener('dragstart', e => {
    e.dataTransfer.setData('text/plain', a.id);
  });

  return el;
}

function folderBox(f){
  const wrap = document.createElement('div');
  wrap.className = 'folder';

  wrap.innerHTML = `
    <div class="body">

      <!-- 🔥 AANGEPAST: OK KNOP BOVEN -->
      <div style="display:flex; justify-content:space-between; align-items:center;">
        <h3>
          <span class="preview">📁</span>
          <span class="pill">${f.icon}</span>
        </h3>
        <button class="okMap btn-small" data-folder="${f.id}">OK</button>
      </div>

      <div class="nameRow">
        <input type="text" placeholder="Typ naam..." data-folder="${f.id}">
        <span id="nameState-${f.id}"></span>
      </div>

      <div class="drop" data-folder="${f.id}"></div>
    </div>
  `;

  const input = wrap.querySelector('input');
  const btn = wrap.querySelector('.okMap');

  btn.onclick = ()=>{
    if(input.value.toLowerCase() === f.name.toLowerCase()){
      input.classList.add('ok');
      input.disabled = true;
      namesOK[f.id] = true;
      checkAllNamed();
    }else{
      input.classList.add('err');
      namesOK[f.id] = false;
    }
  };

  return wrap;
}

function checkAllNamed(){
  allReady = FOLDERS.every(f => namesOK[f.id]);
  if(allReady){
    document.querySelectorAll('.card').forEach(c=>{
      c.classList.remove('disabled');
      c.draggable = true;
    });

    document.querySelectorAll('.drop').forEach(zone=>{
      zone.ondragover = e=> e.preventDefault();
      zone.ondrop = e=>{
        e.preventDefault();
        const id = e.dataTransfer.getData('text/plain');
        const el = document.getElementById('card-'+id);
        zone.appendChild(el);
      };
    });
  }
}

function build(){
  bank.innerHTML='';
  foldersWrap.innerHTML='';

  ANIMALS.forEach(a => bank.appendChild(card(a)));
  FOLDERS.forEach(f => foldersWrap.appendChild(folderBox(f)));

  namesOK = {};
  allReady = false;
}

build();
</script>
